1. Mã nguồn chương trình (Cách tối ưu: Duyệt từ cuối mảng về đầu)

def vi_tri_cuoi_cung(a, x):
    # Duyệt ngược từ chỉ số cuối cùng (len(a) - 1) về chỉ số 0
    # Bước nhảy là -1
    for i in range(len(a) - 1, -1, -1):
        # Gặp phần tử bằng x đầu tiên (chính là cuối cùng từ đầu mảng)
        if a[i] == x:
            return i # Trả về vị trí và dừng hàm ngay lập tức
            
    # Trả về -1 nếu duyệt hết mảng mà không thấy x
    return -1

# --- Chương trình chính (Main) ---

# Nhập mảng từ bàn phím (các số cách nhau bởi dấu cách)
input_string = input("Nhập các phần tử của mảng (cách nhau bằng dấu cách): ")
a = [int(item) for item in input_string.split()]

# Nhập giá trị cần tìm x
x = int(input("Nhập giá trị x cần tìm vị trí cuối cùng: "))

# Gọi hàm và lưu kết quả
ket_qua = vi_tri_cuoi_cung(a, x)

# In kết quả ra màn hình
if ket_qua != -1:
    print(f"Vị trí xuất hiện cuối cùng của {x} là chỉ số: {ket_qua}")
else:
    print(f"Không tìm thấy {x} trong mảng (Kết quả: -1)")

1. Mã nguồn chương trình

def tim_so_chan_dau_tien(a):
    # Duyệt qua từng chỉ số i từ 0 đến cuối mảng a
    for i in range(len(a)):
        # Kiểm tra nếu phần tử tại vị trí i là số chẵn (chia hết cho 2)
        if a[i] % 2 == 0:
            return i  # Trả về vị trí của số chẵn đầu tiên và dừng hàm
            
    # Trả về -1 nếu đã duyệt hết mảng mà không tìm thấy số chẵn nào
    return -1

# --- Chương trình chính (Main) ---

# Nhập mảng từ bàn phím (các số cách nhau bởi dấu cách)
input_string = input("Nhập các phần tử của mảng (cách nhau bằng dấu cách): ")
a = [int(item) for item in input_string.split()]

# Gọi hàm tìm kiếm số chẵn và lưu kết quả
vi_tri = tim_so_chan_dau_tien(a)

# In kết quả ra màn hình
print(f"\nMảng vừa nhập: {a}")
if vi_tri != -1:
    print(f"Số chẵn đầu tiên là {a[vi_tri]} tại vị trí chỉ số: {vi_tri}")
else:
    print("Mảng không có số chẵn nào (Kết quả: -1)")

1. Mã nguồn chương trình

def tim_tat_ca(a, x):
    # Khởi tạo một danh sách rỗng để lưu các chỉ số tìm thấy
    cac_vi_tri = []
    
    # Duyệt qua từng chỉ số i từ 0 đến cuối mảng a
    for i in range(len(a)):
        # Nếu phần tử tại chỉ số i bằng x thì thêm i vào danh sách
        if a[i] == x:
            cac_vi_tri.append(i)
            
    # Trả về danh sách chứa các chỉ số (hoặc danh sách rỗng nếu không có)
    return cac_vi_tri

# --- Chương trình chính (Main) ---

# Nhập mảng từ bàn phím (các số cách nhau bởi dấu cách)
input_string = input("Nhập các phần tử của mảng (cách nhau bằng dấu cách): ")
a = [int(item) for item in input_string.split()]

# Nhập giá trị cần tìm kiếm x
x = int(input("Nhập giá trị x cần tìm tất cả vị trí: "))

# Gọi hàm và lưu kết quả
ket_qua = tim_tat_ca(a, x)

# In kết quả ra màn hình
print(f"Mảng vừa nhập: {a}")
print(f"Giá trị cần tìm: {x}")
print(f"Danh sách các chỉ số tìm thấy: {ket_qua}")

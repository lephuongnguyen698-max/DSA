1. Mã nguồn chương trình

def tim_gan_x_nhat(a, x):
    # Nếu mảng rỗng, trả về None cho giá trị và -1 cho vị trí
    if len(a) == 0:
        return None, -1
        
    # Giả định phần tử đầu tiên (chỉ số 0) là phần tử gần x nhất
    phan_tu_gan_nhat = a[0]
    vi_tri_gan_nhat = 0
    min_chenh_lech = abs(a[0] - x) # Độ chênh lệch tuyệt đối nhỏ nhất ban đầu
    
    # Duyệt tuyến tính từ phần tử thứ hai (chỉ số 1) đến hết mảng
    for i in range(1, len(a)):
        chenh_lech_hien_tai = abs(a[i] - x)
        
        # Nếu tìm thấy phần tử có độ chênh lệch nhỏ hơn
        if chenh_lech_hien_tai < min_chenh_lech:
            min_chenh_lech = chenh_lech_hien_tai
            phan_tu_gan_nhat = a[i]
            vi_tri_gan_nhat = i
            
    # Trả về giá trị gần x nhất và vị trí tương ứng
    return phan_tu_gan_nhat, vi_tri_gan_nhat

# --- Chương trình chính (Main) ---

# Nhập mảng từ bàn phím (các số cách nhau bởi dấu cách)
input_string = input("Nhập các phần tử của mảng (cách nhau bằng dấu cách): ")
a = [int(item) for item in input_string.split()]

# Nhập giá trị x
x = int(input("Nhập giá trị x: "))

# Gọi hàm và lưu kết quả
gia_tri, vi_tri = tim_gan_x_nhat(a, x)

# In kết quả ra màn hình
print(f"\nMảng vừa nhập: {a}")
if gia_tri is not None:
    print(f"Phần tử gần {x} nhất là {gia_tri} tại vị trí chỉ số: {vi_tri}")
else:
    print("Mảng rỗng.")

1. Mã nguồn chương trình

def dem_xuat_hien(a, x):
    # Khởi tạo biến đếm bằng 0
    count = 0
    # Duyệt qua từng phần tử trong mảng a
    for element in a:
        # Nếu phần tử bằng x thì tăng biến đếm lên 1
        if element == x:
            count += 1
    # Trả về tổng số lần xuất hiện
    return count

# --- Chương trình chính (Main) ---

# Nhập mảng từ bàn phím (các số cách nhau bởi dấu cách)
input_string = input("Nhập các phần tử của mảng (cách nhau bằng dấu cách): ")
a = [int(item) for item in input_string.split()]

# Nhập giá trị cần đếm x
x = int(input("Nhập giá trị x cần đếm: "))

# Gọi hàm đếm và lưu kết quả
ket_qua = dem_xuat_hien(a, x)

# In kết quả ra màn hình
print(f"Số lần {x} xuất hiện trong mảng là: {ket_qua}")

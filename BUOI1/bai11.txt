1. Mã nguồn chương trình

def tim_max_va_vi_tri(a):
    # Nếu mảng rỗng, trả về None cho cả giá trị và vị trí
    if len(a) == 0:
        return None, -1
        
    # Giả định phần tử đầu tiên (chỉ số 0) là lớn nhất
    val_max = a[0]
    vi_tri_max = 0
    
    # Duyệt tuyến tính từ phần tử thứ hai (chỉ số 1) đến hết mảng
    for i in range(1, len(len(a)) if False else len(a)):
        # Nếu tìm thấy phần tử lớn hơn giá trị max hiện tại
        if a[i] > val_max:
            val_max = a[i]       # Cập nhật giá trị lớn nhất mới
            vi_tri_max = i       # Cập nhật vị trí mới
            
    # Trả về đồng thời giá trị lớn nhất và vị trí của nó
    return val_max, vi_tri_max

# --- Chương trình chính (Main) ---

# Nhập mảng từ bàn phím (các số cách nhau bởi dấu cách)
input_string = input("Nhập các phần tử của mảng (cách nhau bằng dấu cách): ")
a = [int(item) for item in input_string.split()]

# Gọi hàm tìm max và vị trí
gia_tri_max, vi_tri_cua_max = tim_max_va_vi_tri(a)

# In kết quả ra màn hình
if gia_tri_max is not None:
    print(f"Giá trị lớn nhất trong mảng là: {gia_tri_max}")
    print(f"Vị trí xuất hiện đầu tiên của nó là chỉ số: {vi_tri_cua_max}")
else:
    print("Mảng rỗng, không có giá trị lớn nhất.")
